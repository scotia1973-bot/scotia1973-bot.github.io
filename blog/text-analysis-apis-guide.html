2. Readability Scoring

Readability scores measure how easy a piece of text is to read. The most widely used formula is the Flesch-Kincaid Grade Level, which produces a U.S. school grade level score based on average sentence length and average syllables per word.

Calculating Readability with the Hash & Lorem APIs

While the GadgetHumans API Hub doesn't have a dedicated readability endpoint, you can build a simple readability checker using the Lorem Ipsum API for generating test text and the Hash API for verifying text integrity:

import requests
import re

def readability_grade_level(text):
    """Calculate Flesch-Kincaid Grade Level of a text."""
    sentences = len(re.split(r'[.!?]+', text)) - 1
    words = len(text.split())
    syllables = 0
    
    # Simple syllable counting (approximation)
    for word in text.split():
        word = word.lower().strip('.,!?;:()[]{}"\'')
        if not word:
            continue
        # Count vowel groups
        vowels = len(re.findall(r'[aeiouy]+', word))
        syllables += max(1, vowels)
    
    if sentences == 0 or words == 0:
        return 0
    
    # Flesch-Kincaid Grade Level formula
    grade = 0.39 * (words / sentences) + 11.8 * (syllables / words) - 15.59
    return round(grade, 1)

# Generate some test text from the Lorem API
resp = requests.get(
    "https://api.gadgethumans.com/lorem",
    params={"type": "paragraphs", "count": 3}
)
text = resp.json()["text"]

# Calculate readability
grade = readability_grade_level(text)
print(f"Readability Grade Level: {grade}")
if grade <= 6:
    print("📖 Very easy to read (6th grade level)")
elif grade <= 9:
    print("📖 Moderate (high school level)")
elif grade <= 12:
    print("📖 Fairly difficult (college level)")
else:
    print("📖 Difficult (graduate level)")

While this is a simplified implementation, it demonstrates the concept. For production-grade readability analysis, consider integrating a dedicated NLP library like textstat (Python) alongside the API Hub for other text tasks.

Verifying File Integrity

import requests

def verify_text_integrity(text, expected_hash, algorithm="sha256"):
    """Verify that text matches an expected hash."""
    response = requests.get(
        "https://api.gadgethumans.com/hash",
        params={"text": text, "algorithm": algorithm}
    )
    
    if response.status_code == 200:
        actual_hash = response.json()[algorithm]
        match = actual_hash == expected_hash
        return {
            "match": match,
            "expected": expected_hash,
            "actual": actual_hash
        }
    raise Exception(f"API error: {response.status_code}")

# Example: Verify a known SHA-256 hash
result = verify_text_integrity(
    "hello",
    "2cf24dba5fb0a30e26e83b2ac5b9e29e1b161e5c1fa7425e73043362938b9824"
)
print(f"Hash match: {result['match']}")  # True

Integration with Design Mockups

import requests
from string import Template

def generate_mockup_content(paragraphs=2):
    """Generate realistic placeholder content for a mockup."""
    # Generate paragraphs
    resp = requests.get(
        "https://api.gadgethumans.com/lorem",
        params={"type": "paragraphs", "count": paragraphs}
    )
    body_text = resp.json()["text"]
    
    # Generate a title (short phrase)
    resp = requests.get(
        "https://api.gadgethumans.com/lorem",
        params={"type": "words", "count": 5}
    )
    title = " ".join(w.capitalize() for w in resp.json()["text"].split())
    
    # Generate a subtitle (short sentence)
    resp = requests.get(
        "https://api.gadgethumans.com/lorem",
        params={"type": "sentences", "count": 1}
    )
    subtitle = resp.json()["text"].capitalize()
    
    return {
        "title": title,
        "subtitle": subtitle,
        "body": body_text
    }

5. Building a Text Analysis Pipeline

Let's combine multiple API Hub endpoints into a single text analysis pipeline:

import requests

def analyze_text(text):
    """Run a complete text analysis using the GadgetHumans API Hub."""
    results = {}
    
    # 1. Hash the text for identification
    resp = requests.get(
        "https://api.gadgethumans.com/hash",
        params={"text": text, "algorithm": "sha256"}
    )
    results["sha256"] = resp.json()["sha256"]
    
    # 2. Count basic stats
    words = text.split()
    sentences = text.count('.') + text.count('!') + text.count('?')
    results["word_count"] = len(words)
    results["char_count"] = len(text)
    results["sentence_count"] = sentences
    
    # 3. Generate related test content from the lorem API
    resp = requests.get(
        "https://api.gadgethumans.com/lorem",
        params={"type": "words", "count": 10}
    )
    results["sample_placeholder"] = resp.json()["text"]
    
    # 4. Encode for safe transmission
    resp = requests.get(
        "https://api.gadgethumans.com/base64",
        params={"text": text[:100], "mode": "encode"}
    )
    results["base64_preview"] = resp.json()["output"]
    
    return results

# Analyze a sample text
sample = """The GadgetHumans API Hub provides free developer APIs for common tasks.
You can generate QR codes, passwords, UUIDs, and more with simple HTTP requests.
All endpoints are free up to 100 requests per day with no signup required."""

analysis = analyze_text(sample)
for key, value in analysis.items():
    print(f"{key:20s}: {value}")

This pipeline demonstrates how multiple simple API calls can work together to provide meaningful text analysis — all without installing a single library.
