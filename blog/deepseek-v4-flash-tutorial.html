    Building a Simple AI Chat App in Python

    Now let’s turn that into a Python application. The following script sends a user message, prints the response, and loops for continuous conversation.

    import requests
import json

API_URL = "https://api.gadgethumans.com/chat"
API_KEY = "gh-ds-YOUR_API_KEY_HERE"  # Replace with your actual key

def chat_with_deepseek(messages):
    headers = {
        "Authorization": f"Bearer {API_KEY}",
        "Content-Type": "application/json"
    }
    payload = {
        "model": "deepseek-v4-flash",
        "messages": messages,
        "max_tokens": 500,
        "temperature": 0.7
    }
    response = requests.post(API_URL, headers=headers, json=payload)
    response.raise_for_status()
    return response.json()["choices"][0]["message"]["content"]

def main():
    print("DeepSeek V4 Flash Chat (type 'quit' to exit)")
    messages = [
        {"role": "system", "content": "You are a concise and helpful AI assistant."}
    ]
    while True:
        user_input = input("\nYou: ")
        if user_input.lower() in ("quit", "exit"):
            break
        messages.append({"role": "user", "content": user_input})
        try:
            reply = chat_with_deepseek(messages)
            print(f"Assistant: {reply}")
            messages.append({"role": "assistant", "content": reply})
        except Exception as e:
            print(f"Error: {e}")

if __name__ == "__main__":
    main()

    Save as `chat.py` and run:

    python chat.py
