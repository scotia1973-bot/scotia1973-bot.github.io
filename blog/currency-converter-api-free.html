2. Timestamp Conversion for Date Handling

Python: Timestamp Utility for Rate Tracking

import requests
from datetime import datetime

def get_timestamp_info(timestamp=None):
    """Get information about a Unix timestamp or the current time."""
    params = {}
    if timestamp:
        params["value"] = str(timestamp)
    
    response = requests.get(
        "https://api.gadgethumans.com/timestamp",
        params=params
    )
    return response.json()

# Get info about now
now_info = get_timestamp_info()
print(f"Current time: {now_info['utc']}")
print(f"Day of week: {now_info['day_of_week']}")

# Get info about a specific historical date
rate_date = get_timestamp_info(1700000000)
print(f"Rate timestamp: {rate_date['utc']}")
print(f"Relative time: {rate_date['relative']}")

def detect_user_currency(ip_address=None):
    """Detect the user's likely currency based on their IP address."""
    params = {}
    if ip_address:
        params["ip"] = ip_address
    
    response = requests.get(
        "https://api.gadgethumans.com/ip",
        params=params
    )
    data = response.json()
    
    country = data.get("country", "")
    currency = CURRENCY_MAP.get(country, "USD")  # Default to USD
    
    return {
        "country": country,
        "city": data.get("city", ""),
        "currency": currency,
        "timezone": data.get("timezone", "")
    }

# Detect currency for a US IP

print(f"Location: {result['city']}, {result['country']}")
print(f"Detected currency: {result['currency']}")

# → Detected currency: USD

4. Building a Currency Conversion Dashboard

Let's combine all three APIs into a real-world currency conversion system:

import requests
from datetime import datetime
import json

class CurrencyConverter:
    """Simple currency converter using free APIs."""
    
    # Static exchange rates (for demo — use a real forex API in production)
    RATES = {
        "USD": 1.0, "EUR": 0.92, "GBP": 0.79, "JPY": 149.50,
        "CAD": 1.36, "AUD": 1.53, "CHF": 0.88, "INR": 83.12,
        "CNY": 7.24, "BRL": 5.02, "KRW": 1320.0, "MXN": 17.15,
        "SGD": 1.34, "HKD": 7.82, "SEK": 10.45, "NOK": 10.72,
        "DKK": 6.88
    }
    
    def convert(self, amount, from_currency, to_currency):
        """Convert amount from one currency to another."""
        if from_currency not in self.RATES or to_currency not in self.RATES:
            raise ValueError(f"Unsupported currency")
        
        # Convert to USD first (base), then to target
        usd_amount = amount / self.RATES[from_currency]
        converted = usd_amount * self.RATES[to_currency]
        
        return round(converted, 2)
    
    def get_rates_with_timestamp(self):
        """Get rates with a timestamp from the API."""
        ts = requests.get("https://api.gadgethumans.com/timestamp").json()
        
        return {
            "base": "USD",
            "rates": self.RATES,
            "date": ts["utc"],
            "timestamp": ts["timestamp"]
        }

# Create converter
conv = CurrencyConverter()

# Convert $100 USD to EUR
result = conv.convert(100, "USD", "EUR")
print(f"$100 USD = €{result} EUR")

# Convert $50 USD to JPY
result = conv.convert(50, "USD", "JPY")
print(f"$50 USD = ¥{result} JPY")

# Get full rates with timestamp
rates = conv.get_rates_with_timestamp()
print(f"\nExchange rates as of {rates['date']}:")
for currency, rate in rates["rates"].items():
    if currency != "USD":
        print(f"  USD → {currency}: {rate}")
Important: The exchange rates above are examples for illustration. In production, use a dedicated forex API like ExchangeRate-API, Open Exchange Rates, or Fixer.io for real-time rates. const CURRENCY_RATES = {
    USD: 1.0, EUR: 0.92, GBP: 0.79, JPY: 149.50,
    CAD: 1.36, AUD: 1.53, CHF: 0.88, INR: 83.12,
    CNY: 7.24, BRL: 5.02, KRW: 1320.0, MXN: 17.15
};

async function convertCurrency(amount, from, to) {
    // USD base conversion
    const usdAmount = amount / CURRENCY_RATES[from];
    const converted = usdAmount * CURRENCY_RATES[to];
    return Math.round(converted * 100) / 100;
}

async function getConversionTimestamp() {
    const response = await fetch('https://api.gadgethumans.com/timestamp');
    return response.json();
}

async function getLocalCurrency() {
    const ipResponse = await fetch('https://api.gadgethumans.com/ip');
    const ipData = await ipResponse.json();
    
    const currencyMap = {
        'United States': 'USD', 'United Kingdom': 'GBP',
        'Germany': 'EUR', 'France': 'EUR', 'Japan': 'JPY',
        'Canada': 'CAD', 'Australia': 'AUD'
    };
    
    return currencyMap[ipData.country] || 'USD';
}

// Example usage
(async () => {
    // Get local currency
    const localCurrency = await getLocalCurrency();
    console.log(`Your local currency: ${localCurrency}`);
    
    // Convert $100 USD to local
    const converted = await convertCurrency(100, 'USD', localCurrency);
    console.log(`$100 USD = ${converted} ${localCurrency}`);
    
    // Get timestamp
    const ts = await getConversionTimestamp();
    console.log(`Rate as of: ${ts.utc}`);
})();

5. Best Practices

  • Cache exchange rates: Forex rates don't change minute-to-minute for most use cases. Cache for 1-5 minutes to reduce API calls.
  • Use timestamp for rate versioning: Store the Unix timestamp alongside each rate snapshot so you can show "rates as of" dates.
  • Graceful fallbacks: If the forex API is down, use last known rates with a "stale" indicator.
  • Round appropriately: Always round currency to 2 decimal places for display, but keep full precision for calculations.
